        SICP 1.22 #

        문제 #

        #lang sicp
(define (timed-prime-test n)
  (newline)
  (display n)
  (start-prime-test n (runtime)))

(define (start-prime-test n start-time)
  (if (prime? n)
      (report-prime (- (runtime) start-time))))

(define (report-prime elapsed-time)
  (display " *** ")
  (display elapsed-time))

(define (prime? n)
  (= n (smallest-divisor n)))

(define (smallest-divisor n)
  (find-divisor n 2))

(define (find-divisor n test-divisor)
  (cond ((> (square test-divisor) n) n)
        ((divides? test-divisor n) test-divisor)
        (else (find-divisor n (+ test-divisor 1)))))

(define (divides? a b)
  (= (remainder b a) 0))

(define (square x)
  (* x x))

        정해진 넓이에 속하는 홀수를 차례로 소수인지 알아보는 search-for-primes 프로시저를 짜고 이 프로시저로 1000, 10000, 100000, 1000000 을 실행시켜 소수를 찾는 시간이 루트n 즉 루트10에 수렴하는지 봐라. #

        문제풀이 #

        #lang sicp
(define (timed-prime-test n)
  (newline)
  (display n)
  (start-prime-test n (runtime)))

(define (start-prime-test n start-time)
  (if (prime? n)
      (report-prime (- (runtime) start-time))))

(define (report-prime elapsed-time)
  (display " *** ")
  (display elapsed-time))

(define (prime? n)
  (= n (smallest-divisor n)))

(define (smallest-divisor n)
  (find-divisor n 2))

(define (find-divisor n test-divisor)
  (cond ((> (square test-divisor) n) n)
        ((divides? test-divisor n) test-divisor)
        (else (find-divisor n (+ test-divisor 1)))))

(define (divides? a b)
  (= (remainder b a) 0))

(define (square x)
  (* x x))

(define (even? n)
  (= (remainder n 2) 0))

(define (search-for-primes start count n)
  (cond ((= count n) 0)
        ((prime? start) (timed-prime-test start) (search-for-primes (+ start 1) (+ count 1) n))
        (else (search-for-primes (+ start 1) count n))))
